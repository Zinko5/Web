Instalar Dotnet SDK .NET 6.0
# Instalar .NET 6 wget https://packages.microsoft.com/config/ubuntu/22.04/packages-microsoft-prod.deb -O packages-microsoft-prod.deb
sudo dpkg -i packages-microsoft-prod.deb
rm packages-microsoft-prod.deb # Crear un archivo para priorizar los repositorios PMC sudo cat > /etc/apt/preferences
 # Colocar Package: *
Pin: origin "packages.microsoft.com"
Pin-Priority: 1001
 # Presionar Ctrl + D para terminar

# Instalar el SDK 6.0 de .NET sudo apt update
sudo apt install dotnet-sdk-6.0
